Delete

删除

删除操作提供了不同交互方式组件

  • DeleteButton
  • DeleteWithButton
  • DeleteWithButton
  • DeleteWithConfirmButton

示例

import { List, Datagrid, Column, DeleteButton } from 'prophet-antd';

export default props => (
    <List {...props}>
        <Datagrid>
            <Column>title</Column>
            <Column
                render={record => (
                    <DeleteButton
                        id={record}
                        record={record}
                        resource={props.resource}
                        label="删除"
                    />
                )}
            >
                操作
            </Column>
        </Datagrid>
    </List>
);
